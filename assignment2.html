//create variables for pins
const int red = 3;
const int green = 5;
const int blue = 6;
const int button = 12;


//create variables for rgb color
const int r = 250;
const int g = 0;
const int b = 160;

//create variables for button and led state
int buttonState = 0;
int ledState = 0;

// the setup function runs once when you press reset or power the board
void setup() {
  // initialize pins as outputs or inputs.
  pinMode(red, OUTPUT);
  pinMode(green, OUTPUT);
  pinMode(blue, OUTPUT);
  pinMode(button, INPUT);

  //turns the LED on
  analogWrite(red, r);
  analogWrite(green, g);
  analogWrite(blue, b);
  //change state to match led
  ledState = HIGH;
}

// the loop function runs over and over again forever
void loop() {
  //check the state of button
  buttonState = digitalRead(button);

  //if the button is pressed
  if (buttonState == HIGH){
    //and the led is on
    if (ledState == HIGH) {
      //dim the led
      for (double ledPercentage = 1.00; ledPercentage >= 0.00; ledPercentage -= .05){
         analogWrite(red, r*ledPercentage);
         analogWrite(green, g*ledPercentage);
         analogWrite(blue, b*ledPercentage);
         delay(50);
      }
      //turn the led off
      analogWrite(red, 0);
      analogWrite(green, 0);
      analogWrite(blue, 0);
      //change state to match led
      ledState = LOW;
    } else {
      //brighten the led
      for (double ledPercentage = 0.00; ledPercentage <= 1.00; ledPercentage += .05){
         analogWrite(red, r*ledPercentage);
         analogWrite(green, g*ledPercentage);
         analogWrite(blue, b*ledPercentage);
         delay(50);
      }
      //change state to match led
      ledState = HIGH;
    }
  }
}
