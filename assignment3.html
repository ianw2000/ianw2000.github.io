I used a 10k resistor for the voltage divider because
it matched the range of the photoresistor.


Blue wires go to output pins, while white goes to an
analog in pin. Red is power and black is ground. 

/*
  A3 Input Output
  This project lights up LEDs depending on how much light the
  photoresistor is recieving. When the light is low, the
  yellow light grows brighter before finally the red light
  turns on, singifying darkness.
  
  HCDE 439 Isla Wisemore
*/

//declare variables for pins
const int analogInPin = A0;
const int redPin = 2;
const int yellowPin = 3;
const int greenPin3 = 4;
const int greenPin2 = 5;
const int greenPin1 = 6;

// value read from the photoresistor
int lightValue = 0;
// value output to the yellow LED
int outputValue = 0;

void setup() {
  // initialize serial communications at 9600 bps:
  Serial.begin(9600);
  //initialize output pins
  pinMode(redPin, OUTPUT);
  pinMode(yellowPin, OUTPUT);
  pinMode(greenPin3, OUTPUT);
  pinMode(greenPin2, OUTPUT);
  pinMode(greenPin1, OUTPUT);

}

void loop() {
  // read the analog in value
  lightValue = analogRead(analogInPin);
  // constrain to clean up data on low and high ends
  lightValue = constrain(lightValue, 300, 900);
  // if value is in lowest bracket
  if (lightValue < 420){
    // turn on red, turn off others
    digitalWrite(redPin, HIGH);
    analogWrite(yellowPin, 0);
    digitalWrite(greenPin3, LOW);
    digitalWrite(greenPin2, LOW);
    digitalWrite(greenPin1, LOW);
    // else if value is in next bracket
  } else if (lightValue < 540){
    // turn on yellow, turn off others
    digitalWrite(redPin, LOW);
    // map the bracket to high->low
    outputValue = map(lightValue, 420, 539, 255, 0);
    analogWrite(yellowPin, outputValue);
    digitalWrite(greenPin3, LOW);
    digitalWrite(greenPin2, LOW);
    digitalWrite(greenPin1, LOW);
    // else if value is in next bracket
  } else if (lightValue < 660){
    // turn on 3rd green pin, turn off others
    digitalWrite(redPin, LOW);
    analogWrite(yellowPin, 0);
    digitalWrite(greenPin3, HIGH);
    digitalWrite(greenPin2, LOW);
    digitalWrite(greenPin1, LOW);
    // else if value is in next bracket
  } else if (lightValue < 780){
    // turn on 2nd green pin, turn off others
    digitalWrite(redPin, LOW);
    analogWrite(yellowPin, 0);
    digitalWrite(greenPin3, LOW);
    digitalWrite(greenPin2, HIGH);
    digitalWrite(greenPin1, LOW);
    // else the value is in the final bracket
  } else {
    // turn on 3rd green pin, turn off others
    digitalWrite(redPin, LOW);
    analogWrite(yellowPin, 0);
    digitalWrite(greenPin3, LOW);
    digitalWrite(greenPin2, LOW);
    digitalWrite(greenPin1, HIGH);
  }

  // wait 2 milliseconds before the next loop for the analog-to-digital
  // converter to settle after the last reading:
  delay(2);
}


The lights correspond to the light level picked up
by the photoresistor, with level yellow getting brighter
as the light level decreases to red.
