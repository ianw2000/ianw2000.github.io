//include both dht and lcd libaries
#include <dht.h>
#include <LiquidCrystal.h>

//initialize an lcd with the correct pins
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

//initialize temp/humidity sensor
dht DHT;

//define the pin connected to dht
#define DHT11_PIN 7

void setup(){
  //start the lcd with the correct rows/columns
  lcd.begin(16, 2);
}

void loop(){
  //check the dht reading
  int chk = DHT.read11(DHT11_PIN);
  //move to the first row
  lcd.setCursor(0,0); 
  //print the temperature
  lcd.print("Temp: ");
  lcd.print(DHT.temperature);
  lcd.print((char)223);
  lcd.print("C");
  //move to the second row
  lcd.setCursor(0,1);
  //print the humidity
  lcd.print("Humidity: ");
  lcd.print(DHT.humidity);
  lcd.print("%");
  //wait
  delay(1000);
}


The display shows the temperature and humidity
measured by the sensor
